module alu (
    input [7:0] A, B,
    input [2:0] Sel,
    output reg [7:0] Result
);

always @(*) begin
    case(Sel)
        3'b000: Result = A + B;   // ADD
        3'b001: Result = A - B;   // SUB
        3'b010: Result = A & B;   // AND
        3'b011: Result = A | B;   // OR
        3'b100: Result = A ^ B;   // XOR
        3'b101: Result = ~A;      // NOT A
        default: Result = 8'b00000000;
    endcase
end

endmodule
